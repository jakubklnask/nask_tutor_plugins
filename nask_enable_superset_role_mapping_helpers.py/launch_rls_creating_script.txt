from superset.extensions import db, security_manager
from superset.connectors.sqla.models import RowLevelSecurityFilter

session = security_manager.get_session()

def clear_all_rls_filters():
    print("Rozpoczynam usuwanie wszystkich filtrów RLS i kluczy obcych...")
    try:
        # Wyciągamy obiekty tabel prosto z metadanych, zero zgadywania importów
        tabela_roles = db.metadata.tables['rls_filter_roles']
        tabela_tables = db.metadata.tables['rls_filter_tables']
        
        # 1. Usuwamy powiązania z rolami i tabelami
        session.execute(tabela_roles.delete())
        session.execute(tabela_tables.delete())
        
        # 2. Usuwamy główne filtry
        liczba_usunietych = session.query(RowLevelSecurityFilter).delete()
        
        session.commit()
        print(f"Sukces: Usunięto {liczba_usunietych} filtrów RLS.")
    except Exception as e:
        session.rollback()
        print(f"Błąd: {e}")

clear_all_rls_filters()
